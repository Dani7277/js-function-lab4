JavaScript Code Overview

// Part 1: Simple function
function greet(name) {
    return `Hello, ${name}!`;
}

// Part 2: Function with parameters and return value
function addNumbers(num1, num2) {
    return num1 + num2;
}

// Part 3: Function scope example
let x = 10;

function changeValue() {
    let x = 20; // This creates a new local variable
    console.log("Inside function, x =", x);
}

// Part 4: Closure example
function outerFunction() {
    let count = 0;
    return function() {
        count++;
        return count;
    };
}

// Bonus: Default parameter
function greetWithDefault(name = "Guest") {
    return `Hello, ${name}!`;
}
